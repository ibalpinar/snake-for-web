// Game Loop with 60 FPS Animation
function main(currentTime) {
  if (gameOver) {
    if (window.saveScore) {
      window.saveScore(score);
    }
    if (confirm('Game Over!\nScore: ' + score + '\nDo you want to try again?')) {
      window.location = './';
    }
    return;
  }

  window.requestAnimationFrame(main);
  const deltaTime = currentTime - lastRenderTime;
  
  // Frame rate control for smooth animation
  accumulatedTime += deltaTime;
  
  if (accumulatedTime > FRAME_TIME) {
    const steps = Math.floor(accumulatedTime / FRAME_TIME);
    accumulatedTime = accumulatedTime % FRAME_TIME;
    
    // Update game state
    for (let i = 0; i < steps; i++) {
      update();
    }
  }
  
  // Interpolation for smooth rendering
  const alpha = accumulatedTime / FRAME_TIME;
  draw(alpha);
  lastRenderTime = currentTime;
}
